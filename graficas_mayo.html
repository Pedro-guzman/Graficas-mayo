# Convertimos la columna Fecha al formato correcto
df['Fecha'] = pd.to_datetime(df['Fecha'], errors='coerce')

# Filtramos solo los servicio de Mayo
df_mayo = df[df['Fecha'].dt.month == 5]

# Agrupamos por peluquero y contamos los servicios
